import java.util.Scanner;
public class StackQuestion1
{
	static int size,top=-1;
	static Scanner input = new Scanner(System.in);
	public static void main(String args[])
	{
		String choice;
		System.out.print("Size of stack : ");
		size = input.nextInt();
		int a[] = new int[size];
		do
		{
			System.out.print("1.PUSH  |  2.POP  |  3.MAX_VALUE |  anyKey=TERMINATE  >> ");
			choice = input.next();
			switch(choice)
			{
				case "1" : push(a);break;
				case "2" : pop(a);break;
				case "3" : max(a);break;
				default : System.out.println("Exited");
			}
		}while(choice.equals("1") ||choice.equals("2") || choice.equals("3"));
	}
	public static void push(int a[])
	{
		System.out.print("Push element : ");
		int item = input.nextInt();
		if(top<size-1)
		{
			top++;a[top]=item;
		}
		else
			System.out.println("Oops! Stack Overflow :(");
	}
	public static void pop(int a[])
	{
		if(top>=0)
		{
			top--;
		}
		else
			System.out.println("Oops! Stack Underflow :(");
	}
	public static void max(int a[])
	{
		if(top>=0)
		{
			int max=0;
			for(int i=0;i<top;i++)
			{
				max = a[i];
				if(max<a[i+1])
				{
					max = a[i+1];
				}
			}
			System.out.println("Maximum value : "+max);
		}
		else
			System.out.println("Oops! Stack Underflow :(");
	}
}
